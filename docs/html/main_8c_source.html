25// Helper pour gérer le chemin (ajoute data/ si nécessaire)
26static void formatter_chemin(const char *entree, char *sortie) {
27 // Si le chemin commence par . ou / ou data/, on touche pas
28 if (entree[0] == '.' || entree[0] == '/' ||
29 strncmp(entree, "data/", 5) == 0) {
30 strcpy(sortie, entree);
31 } else {
32 sprintf(sortie, "data/%s", entree);
33 }
34}

159int main(int argc, char *argv[]) {
160 Graphe *g = NULL;
161 char filepath[256];
162
163 // Initialiser le système de logging
164 logger_init_simple("logs/gsm.log", LOG_LEVEL_INFO);
165 LOG_INFO(MOD_MAIN, "Démarrage de l'interface interactive GSM");
166
167 // Si un fichier est passé en argument, on essaie de le charger direct
168 if (argc > 1) {
169 LOG_INFO(MOD_LOADER, "Chargement depuis argument: %s", argv[1]);
170 g = graphe_charger(argv[1]);
171 if (g) {
172 LOG_INFO(MOD_LOADER, "Graphe chargé avec succès");
173 printf("Graphe charge avec succes !\n");
174 graphe_afficher(g);
175 } else {
176 LOG_ERROR(MOD_LOADER, "Échec du chargement: %s", argv[1]);
177 printf("Echec du chargement.\n");
178 }
179 }
180
181 int choix = -1;
182 while (choix != 0) {
183 afficher_menu();
184 if (scanf("%d", &choix) != 1) {
185 // Vider le buffer si l'utilisateur rentre n'importe quoi
186 while (getchar() != '\n')
187 ;
188 continue;
189 }
190
191 switch (choix) {
192 case 1: // Charger
193 printf("Nom du fichier (dans data/) ou chemin complet : ");
194 scanf("%s", filepath);
195 char fullpath[256];
196 formatter_chemin(filepath, fullpath);
197
198 if (g)
199 graphe_detruire(g); // Liberer l'ancien
200 g = graphe_charger(fullpath);
201 if (g)
202 printf(">>> Graphe charge depuis %s.\n", fullpath);
203 else
204 printf(">>> Erreur de chargement (%s).\n", fullpath);
205 break;
206
207 case 2: // Afficher
208 if (g)
209 graphe_afficher(g);
210 else
211 printf(">>> Aucun graphe en memoire.\n");
212 break;
213
214 case 3: // Dijkstra
215 menu_dijkstra(g);
216 break;
217
218 case 4: // K-Shortest
219 menu_k_shortest(g);
220 break;
221
222 case 5: // Backtracking
223 menu_backtracking(g);
224 break;
225
226 case 6: // Securite Globale
227 if (g)
228 securite_analyser_vulnerabilites(g);
229 else
230 printf(">>> Aucun graphe en memoire.\n");
231 break;
232
233 case 7: // Detecter Cycles
234 if (g) {
235 int has_cycle = detecter_cycles(g);
236 if (has_cycle)
237 printf(">>> ATTENTION: Le graphe contient des cycles !\n");
238 else
239 printf(">>> Le graphe est acyclique (DAG).\n");
240 } else {
241 printf(">>> Aucun graphe en memoire.\n");
242 }
243 break;
244
245 case 8: // Sauvegarder
246 if (!g) {
247 printf(">>> Aucun graphe a sauvegarder.\n");
248 } else {
249 printf("Nom du fichier de sortie (sera dans data/) : ");
250 scanf("%s", filepath);
251 char outpath[256];
252 formatter_chemin(filepath, outpath);
253
254 if (graphe_sauvegarder(g, outpath)) {
255 printf(">>> Topologie sauvegardee dans %s\n", outpath);
256 } else {
257 printf(">>> Erreur lors de la sauvegarde.\n");
258 }
259 }
260 break;
261
262 case 0:
263 LOG_INFO(MOD_MAIN, "Fermeture demandée par l'utilisateur");
264 printf("Au revoir !\n");
265 break;
266
267 default:
268 LOG_WARN(MOD_MAIN, "Choix invalide: %d", choix);
269 printf("Choix invalide.\n");
270 }
271 }
272
273 // Nettoyage
274 if (g) {
275 LOG_DEBUG(MOD_MAIN, "Libération du graphe en mémoire");
276 graphe_detruire(g);
277 }
278
279 LOG_INFO(MOD_MAIN, "Session terminée");
280 logger_close();
281
282 return 0;
283}
