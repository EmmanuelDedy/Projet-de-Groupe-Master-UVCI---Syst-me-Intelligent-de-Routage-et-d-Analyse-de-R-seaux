52/**
53 * @brief Algorithme de Dijkstra (Routage Optimal).
54 *
55 * Trouve le chemin minimisant le critère donné (cout ou latence).
56 * Utilise une stratégie gloutonne par relaxation.
57 *
58 * **Pseudo-code :**
59 * @code
60 * Pour v dans V: dist[v] = INF, pred[v] = NULL
61 * dist[source] = 0
62 * Q = PriorityQueue(V)
63 * Tant que Q non vide:
64 * u = ExtractMin(Q)
65 * Pour chaque voisin v de u:
66 * alt = dist[u] + poids(u,v)
67 * Si alt < dist[v]:
68 * dist[v] = alt
69 * pred[v] = u
70 * Update(Q, v, alt)
71 * @endcode
72 *
73 * @param g Le graphe
74 * @param source ID nœud départ
75 * @param destination ID nœud arrivée
76 * @param critere "cout" ou "latence"
77 * @return Chemin* Structure allouée ou NULL si inaccessible.
78 */
79Chemin *routage_dijkstra(Graphe *g, int source, int destination,
80 const char *critere);
