33/**
34 * @struct Metriques
35 * @brief Vecteur de poids multidimensionnel pour une arête.
36 *
37 * Contrairement à un graphe simple où poids = distance, ici nous gérons 5
38 * dimensions :
39 * - **Cout** : Opex/Capex ou distance physique.
40 * - **Latence** : Délai de propagation (ms).
41 * - **Bande Passante** : Capacité du lien (Mbps).
42 * - **Fiabilité** : Probabilité de panne (0.0 - 1.0).
43 * - **Sécurité** : Niveau de confiance (int).
44 */
45typedef struct {
46 float cout; /**< Coût financier ou distance (km) */
47 float latence; /**< Latence (ms) - Critique pour VoIP */
48 float bande_passante; /**< Bande passante (Mbps) - Critique pour Streaming */
49 float fiabilite; /**< Indice de fiabilité (0-10) ou probabilité */
50 int securite; /**< Niveau de sécurité (1-5) */
51} Metriques;
52
53/**
54 * @struct Noeud
55 * @brief Représentation d'un sommet du graphe (Équipement).
56 */
57typedef struct Noeud {
58 int id; /**< Identifiant unique (Index dans le tableau) */
59 char nom[50]; /**< Nom mnémonique aka Hostname */
60 TypeNoeud type; /**< Rôle de l'équipement */
61 double x, y; /**< Coordonnées géographiques (Visualisation) */
62 bool est_actif; /**< État opérationnel (True=UP, False=DOWN) */
63 struct AreteNoeud *liste_adj; /**< Tête de la liste d'adjacence (Voisins) */
64} Noeud;

66/**
67 * @struct AreteNoeud
68 * @brief Élément de liste chaînée représentant un arc sortant.
69 *
70 * Structure optimisée pour les graphes creux (Sparse Graphs) où |E| << |V|^2.
71 * Chaque nœud possède une liste de ces structures.
72 */
73typedef struct AreteNoeud {
74 int dest_id; /**< ID du nœud de destination */
75 Metriques metriques; /**< Poids de l'arc */
76 struct AreteNoeud *suivant; /**< Pointeur vers le prochain voisin */
77} AreteNoeud;

79/**
80 * @struct Graphe
81 * @brief Structure principale encapsulant la topologie réseau.
82 *
83 * Combine deux représentations pour la performance :
84 * 1. **Tableau de Nœuds** : Accès O(1) aux métadonnées des sommets.
85 * 2. **Matrice d'Adjacence** (Optionnelle) : Pour les graphes denses.
86 * 3. **Listes d'Adjacence** : Pour le parcours efficace des voisins.
87 */
88typedef struct {
89 int nb_noeuds; /**< Nombre actuel de sommets */
90 int max_noeuds; /**< Capacité maximale allouée */
91 Noeud *noeuds; /**< Tableau dynamique des sommets */
92 double **matrice_adj; /**< Matrice d'adjacence (Poids simple ou NULL) */
93 bool est_oriente; /**< Type de graphe (Orienté ou Non) */
94} Graphe;
