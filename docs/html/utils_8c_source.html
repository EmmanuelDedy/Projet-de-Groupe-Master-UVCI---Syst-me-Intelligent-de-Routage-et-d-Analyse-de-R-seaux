35void json_escape_string(const char *input, char *output, size_t max_len) {
36 if (!input || !output || max_len == 0)
37 return;
38
39 size_t j = 0;
40 for (size_t i = 0; input[i] != '\0' && j < max_len - 1; i++) {
41 char c = input[i];
42 if (c == '"' || c == '\\') {
43 if (j + 2 >= max_len)
44 break;
45 output[j++] = '\\';
46 output[j++] = c;
47 } else if (c == '\n') {
48 if (j + 2 >= max_len)
49 break;
50 output[j++] = '\\';
51 output[j++] = 'n';
52 } else if (c == '\r') {
53 if (j + 2 >= max_len)
54 break;
55 output[j++] = '\\';
56 output[j++] = 'r';
57 } else if (c == '\t') {
58 if (j + 2 >= max_len)
59 break;
60 output[j++] = '\\';
61 output[j++] = 't';
62 } else {
63 output[j++] = c;
64 }
65 }
66 output[j] = '\0';
67}

129bool ensure_directory(const char *filepath) {
130 if (!filepath)
131 return false;
132
133 // Extraire le répertoire parent
134 char dir[512];
135 strncpy(dir, filepath, sizeof(dir) - 1);
136 dir[sizeof(dir) - 1] = '\0';
137
138 // Trouver le dernier séparateur
139 char *last_sep = strrchr(dir, '/');
140 if (!last_sep)
141 return true; // Pas de répertoire parent
142
143 *last_sep = '\0'; // Tronquer au répertoire parent
144
145 // Créer récursivement (simple, un seul niveau)
146 struct stat st;
147 if (stat(dir, &st) != 0) {
148#ifdef _WIN32
149 return mkdir(dir) == 0;
150#else
151 return mkdir(dir, 0755) == 0;
152#endif
153 }
154 return true;
155}

161void utils_generer_topologie_large(const char *filename, const char *name,
162 int nb_noeuds, int density_percent) {
163 // Ensure we have a random seed if not already set (call srand in main
164 // ideally) srand(time(NULL));
165
166 // Ensure directory exists
167 ensure_directory(filename);
168
169 FILE *f = fopen(filename, "w");
170 if (!f) {
171 perror("utils_generer_topologie_large fopen");
172 return;
173 }
174
175 // Header: N <nb> <oriente=0>
176 fprintf(f, "N %d 0\n", nb_noeuds);
177
178 // Nodes: V <id> <name> <type> <x> <y> <active>
179 for (int i = 0; i < nb_noeuds; i++) {
180 int type = rand() % 4; // 0=ROUTEUR, 1=COMMUTATEUR...
181 float x = (float)(rand() % 1000);
182 float y = (float)(rand() % 1000);
183 fprintf(f, "V %d Node_%d %d %.2f %.2f 1\n", i, i, type, x, y);
184 }
185
186 // Edges: E <src> <dest> <cost> <lat> <bp> <fiab> <sec>
187 for (int i = 0; i < nb_noeuds; i++) {
188 for (int j = i + 1; j < nb_noeuds; j++) {
189 if ((rand() % 100) < density_percent) {
190 float cout = (float)(rand() % 20 + 1);
191 float lat = (float)(rand() % 10 + 1);
192 float bp = 1000.0f;
193 float fiab = 0.9f + ((rand() % 10) / 100.0f); // 0.90 - 0.99
194 int sec = (rand() % 3) + 1; // 1-3
195
196 fprintf(f, "E %d %d %.2f %.2f %.2f %.2f %d\n", i, j, cout, lat, bp,
197 fiab, sec);
198 }
199 }
200 // Ensure connectivity ring/chain
201 if (i < nb_noeuds - 1) {
202 fprintf(f, "E %d %d 10.00 1.00 1000.00 0.99 1\n", i, i + 1);
203 }
204 }
205 fclose(f);
206 printf("[Utils] Generated topology '%s' to %s\n", name, filename);
207}
