50Chemin *routage_dijkstra(Graphe *g, int source, int destination,
51 const char *critere) {
52 if (!g || source < 0 || source >= g->max_noeuds)
53 return NULL;
54
55 int n = g->max_noeuds;
56 float *dist = (float *)malloc(n * sizeof(float));
57 int *pred = (int *)malloc(n * sizeof(int));
58 bool *visite = (bool *)calloc(n, sizeof(bool));
59
60 for (int i = 0; i < n; i++) {
61 dist[i] = FLT_MAX;
62 pred[i] = -1;
63 }
64 dist[source] = 0.0f;
65
66 for (int i = 0; i < n; i++) {
67 int u = -1;
68 float min_d = FLT_MAX;
69 for (int v = 0; v < n; v++) {
70 if (!visite[v] && dist[v] < min_d) {
71 min_d = dist[v];
72 u = v;
73 }
74 }
75 if (u == -1 || u == destination)
76 break;
77 visite[u] = true;
78
79 AreteNoeud *a = g->noeuds[u].liste_adj;
80 while (a) {
81 int v = a->dest_id;
82 if (!visite[v] && g->noeuds[v].est_actif) {
83 float p = obtenir_poids(a, critere);
84 if (dist[u] + p < dist[v]) {
85 dist[v] = dist[u] + p;
86 pred[v] = u;
87 }
88 }
89 a = a->suivant;
90 }
91 }
92
93 Chemin *res = NULL;
94 if (destination >= 0 && destination < n && dist[destination] != FLT_MAX) {
95 res = (Chemin *)malloc(sizeof(Chemin));
96 int curr = destination, len = 0;
97 while (curr != -1) {
98 len++;
99 curr = pred[curr];
100 }
101 res->longueur = len;
102 res->noeuds = (int *)malloc(len * sizeof(int));
103 curr = destination;
104 for (int i = len - 1; i >= 0; i--) {
105 res->noeuds[i] = curr;
106 curr = pred[curr];
107 }
108
109 // Recalcul des métriques réelles du trajet
110 res->cout_total = 0;
111 res->latence_totale = 0;
112 res->bande_passante_min = FLT_MAX;
113 res->securite_min = 10;
114 for (int i = 0; i < len - 1; i++) {
115 AreteNoeud *a = g->noeuds[res->noeuds[i]].liste_adj;
116 while (a && a->dest_id != res->noeuds[i + 1])
117 a = a->suivant;
118 if (a) {
119 res->cout_total += a->metriques.cout;
120 res->latence_totale += a->metriques.latence;
121 if (a->metriques.bande_passante < res->bande_passante_min)
122 res->bande_passante_min = a->metriques.bande_passante;
123 if (a->metriques.securite < res->securite_min)
124 res->securite_min = a->metriques.securite;
125 }
126 }
127 }
128 free(dist);
129 free(pred);
130 free(visite);
131 return res;
132}

151static bool chemin_existe_deja(EnsembleChemins *ec, Chemin *c) {
152 for (int i = 0; i < ec->nombre; i++) {
153 if (ec->chemins[i].longueur != c->longueur)
154 continue;
155 bool t = true;
156 for (int j = 0; j < c->longueur; j++) {
157 if (ec->chemins[i].noeuds[j] != c->noeuds[j]) {
158 t = false;
